class MyList(list):
    """
    >>> my_list = mylist()
    >>> my_list.append(3)
    >>> my_list.append(1)
    >>> my_list.append(4)
    >>> my_list.append(2)
    >>> my_list.print_sorted()
    [1, 2, 3, 4]
    >>> my_list.append(5)
    >>> my_list.print_sorted()
    [1, 2, 3, 4, 5]
    """

   def print_sorted(self):
   """
   print the list in ascending order.
   

   >>> my_list = MyList([3, 1, 4, 2])
   >>> my_list.print_sorted()
   [1, 2, 3, 4]
   """

   sorted_list = sorted(self)
   print(sorted_list)
